sequenceDiagram

    participant Sistema
    participant BD as "Banco de Dados"

    Sistema->>BD: buscarInformacoes(restaurante)
    BD-->>Sistema: informacoesRestaurante
    Sistema->>Cliente: exibirInformacoes(horarios, disponibilidade)
    Cliente->>Sistema: escolherDetalhes(data, horario, pessoas)
    Sistema->>BD: verificarDisponibilidade(data, horario, pessoas)
    BD-->>Sistema: mesasDisponiveis
    Sistema->>Cliente: exibirResumo(reserva)
    Cliente->>Sistema: confirmarReserva()
    Sistema->>BD: registrarReserva(reserva)
    BD->>BD: criarReserva()
    BD-->>Sistema: reservaCriada
    Sistema->>Sistema: gerarCodigo()
    Sistema->>Cliente: exibirCodigoConfirmacao(codigo)
